namespace Mail

Provided By: cmail.fec

The holding namespace for Mail related tools

Example

uses "cmail";
uses "console";

function fetchmail( string host, string username, string password ) {
    object connection = new Mail.Connection();

    if( connection.openMailbox( "{$host:993/imap/ssl/novalidate-cert/user=\"$username\"}", password) ) {
        number count = connection.messageCount();
        for( number i = 1; i <= count; i++ ) {
            object message = connection.messageAtIndex(i);
            Console.println("Got message: ${message.getSubject()}: ${message.toText()}");
        }
    }
}
